    companion object {
        // For Singleton instantiation.
        @Volatile
        private var browserManager :MediaBrowserManager ?=null

        fun getInstance() = browserManager ?: synchronized(this) {
            browserManager?: MediaBrowserManager()
                .also { browserManager = it }
        }
    }
